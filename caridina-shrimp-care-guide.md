    <div class="faq-item" id="white-ring">
      <button class="faq-question" aria-expanded="false" aria-controls="white-ring-answer">

      </div>
    </div>

    <div class="faq-item" id="die-after-molt">
      <button class="faq-question" aria-expanded="false" aria-controls="die-after-molt-answer">

      </button>
      <div class="faq-answer" id="die-after-molt-answer">

      </div>
    </div>

    <div class="faq-item" id="healthy">
      <button class="faq-question" aria-expanded="false" aria-controls="healthy-answer">

      </button>
